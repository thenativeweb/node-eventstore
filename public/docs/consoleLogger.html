    log('info', 'green', msg);

logger.warn = function(msg) {
    log('warn', 'yellow', msg);
};
    
logger.error = function(msg) {
    log('error', 'red', msg);
};


function log(prefix, style, msg) {
    console.log(stylize(prefix, style) + ': ' + new Date().toLocaleString() + ': ' + msg);
}
